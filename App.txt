            <p class="text-gray-500">আপনার নামকে দিন একটি নতুন লুক!</p>

        <!-- Input Section -->
        <div class="mb-6">
            <label class="block text-sm font-semibold text-gray-700 mb-2">আপনার নাম লিখুন:</label>
            <input type="text" id="nameInput" placeholder="উদাঃ Ariful" 
                class="w-full px-4 py-3 rounded-xl border border-gray-300 focus:ring-2 focus:ring-indigo-500 focus:outline-none transition-all text-lg">
        </div>

        <!-- Styles Container -->
        <div id="results" class="space-y-3">
            <!-- Styles will be generated here -->
            <div class="text-center text-gray-400 py-10">
                <i class="fas fa-magic text-4xl mb-3"></i>
                <p>উপরে নাম লিখলে স্টাইল দেখা যাবে</p>
            </div>
        </div>
    </div>

    <!-- Toast Notification -->
    <div id="toast" class="toast">কপি করা হয়েছে!</div>

    <script>
        const nameInput = document.getElementById('nameInput');
        const results = document.getElementById('results');
        const toast = document.getElementById('toast');

        const fontStyles = [
            (t) => `꧁༺ ${t} ༻꧂`,
            (t) => `★彡 ${t} 彡★`,
            (t) => `『 ${t} 』`,
            (t) => `亗 ${t} 亗`,
            (t) => `⫷ ${t} ⫸`,
            (t) => `〆 ${t} 〆`,
            (t) => `『V』 ${t}`,
            (t) => `亗 ${t} FF 亗`,
            (t) => `⚡ ${t} ⚡`,
            (t) => `❤ ${t} ❤`,
            (t) => `꧁ ${t} ꧂`,
            (t) => `(✿ ${t} ✿)`,
            (t) => `♛ ${t} ♛`,
            (t) => `【 ${t} 】`,
            (t) => `彡 ${t} 彡`,
            (t) => `░ ${t} ░`,
            (t) => `⚔ ${t} ⚔`
        ];

        function showToast() {
            toast.style.display = 'block';
            setTimeout(() => {
                toast.style.display = 'none';
            }, 2000);
        }

        function copyToClipboard(text) {
            const el = document.createElement('textarea');
            el.value = text;
            document.body.appendChild(el);
            el.select();
            document.execCommand('copy');
            document.body.removeChild(el);
            showToast();
        }

        function generateNames() {
            const val = nameInput.value.trim();
            if (!val) {
                results.innerHTML = `
                    <div class="text-center text-gray-400 py-10">
                        <i class="fas fa-magic text-4xl mb-3"></i>
                        <p>উপরে নাম লিখলে স্টাইল দেখা যাবে</p>
                    </div>`;
                return;
            }

            results.innerHTML = '';
            fontStyles.forEach(styleFn => {
                const styledName = styleFn(val);
                const div = document.createElement('div');
                div.className = 'name-card flex justify-between items-center bg-white p-4 rounded-xl';
                div.innerHTML = `
                    <span class="text-gray-800 font-medium overflow-hidden truncate mr-2">${styledName}</span>
                    <button class="copy-btn active:scale-95 transition-transform" onclick="copyToClipboard('${styledName}')">
                        <i class="far fa-copy mr-1"></i> Copy
                    </button>
                `;
                results.appendChild(div);
            });
        }

        nameInput.addEventListener('input', generateNames);
    </script>
